# 7. 编程式路由

# 1. 传递params参数并实现跳转

import {useNavigate} from "react-router-dom";

function HomeMessage() {
  const state = {
    messages: [
      {id: '01', title: '消息1'},
      {id: '02', title: '消息2'},
      {id: '03', title: '消息3'},
    ]
  }
  const navigate = useNavigate()
  const toParams = (item) => {
    navigate(`/home/message/detail/${item.id}/${item.title}`,{
    	replace:false,
    })
  }
    return (
      <div>
        <ul>
          {
            state.messages.map(item => {
              return (
                <li key={item.id} >
                  <button onClick={() => toParams(item)}>push</button>
                  <button onClick={() => toParams(item)}>replace</button>
                </li>
              )
            })
          }
        </ul>
        <Routes>
          <Route path="/detail/:id/:title" element={<Detail/>} />
        </Routes>
      </div>
    )
}

export default HomeMessage

# 3. 传递state参数并实现跳转

import {useNavigate} from "react-router-dom";

function HomeMessage() {
  const state = {
    messages: [
      {id: '01', title: '消息1'},
      {id: '02', title: '消息2'},
      {id: '03', title: '消息3'},
    ]
  }
  const navigate = useNavigate()
  const toState = (item) => {
    navigate('/home/message/detail', {
      // push/replace切换  
      replace: false,
      state: {
        id: item.id,
        title: item.title
      }
    })
  }
    return (
      <div>
        <ul>
          {
            state.messages.map(item => {
              return (
                <li key={item.id} >
                  <button onClick={() => toState(item)}>push</button>
                  <button onClick={() => toState(item)}>replace</button>
                </li>
              )
            })
          }
        </ul>
        <Routes>
          <Route path="/detail" element={<Detail/>} />
        </Routes>
      </div>
    )
}

export default HomeMessage

# 8. BrowserRouter / HashRouter

# 1. 底层原理

BrowserRouter使用的是H5的history API，不兼容IE9及以下版本

HashRouter使用的是URL的哈希值

# 2. path表现形式

BrowserRouter的路径中没有#，例如：localhost:3000/demo/test

HashRouter的路径包含#，例如：localhost:3000/#/demo/test

# 3. 刷新后对路由state参数的影响

BrowserRouter没有任何影响，因为state保存在history对象中

HashRouter刷新后导致路由state参数的丢失

# 4. HashRouter 可以用于解决一些路径错误相关的问题
