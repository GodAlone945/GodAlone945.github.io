# 百度地图点击获取地址

Created: July 17, 2021 10:50 PM
标签: JS, 百度地图, 获取地址

```jsx
// 百度地图API功能
var _this = this;
// 创建地图
var map = new BMap.Map('allmap'); // 在指定区域创建地图实例
map.centerAndZoom(new BMap.Point(118.147593, 24.526786), 15); // 对地图进行初始化
map.enableScrollWheelZoom(true);  // 启用滚轮放大缩小
var geoc = new BMap.Geocoder();   // 创建一个地址解析器的实例
map.addEventListener('click', function(e) {
  map.clearOverlays() // 在点击时，清除上次的覆盖物
  // console.log(e);
  var pt = e.point;
  var marker = new BMap.Marker(new BMap.Point(pt.lng, pt.lat)); // 创建一个图像标注实例
  map.addOverlay(marker); // 讲覆盖物添加到地图中，一个覆盖物实例只能向地图中添加一次
  geoc.getLocation(pt, rs => {
    // console.log(rs)
    _this.address = rs.surroundingPois[0].address
  });
});
```
