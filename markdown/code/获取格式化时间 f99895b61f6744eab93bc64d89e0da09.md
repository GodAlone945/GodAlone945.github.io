# 获取格式化时间

Created: October 20, 2021 2:20 PM
标签: JS, 格式化时间

```jsx
function getCurrentFormatDate() {
	// 系统当前时间格式化
	let currentFormatDate = ''
	// 获取系统当前日期
	const date = new Date()
	// 获取当前年
	const currentYear = date.getFullYear()
	// 获取当前月
	let currentMonth = date.getMonth() + 1
	currentMonth = (currentMonth <= 9) ? '0' + currentMonth : currentMonth
	// 获取当前日
	let currentDay = date.getDate()
	currentDay = (currentDay <= 9) ? '0' + currentDay : currentDay
	// 时
	const currentHours = date.getHours()
	// 分
	const currentMinutes = date.getMinutes()
	// 秒
	const currentSeconds = date.getSeconds()
	// yyyy-MM-dd HH:mm:ss
	currentFormatDate = currentYear + '-' + currentMonth + '-' + currentDay + ' ' + currentHours + ':' + currentMinutes + ':' + currentSeconds
	return currentFormatDate
}
```
