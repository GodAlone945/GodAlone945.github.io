```css
    background: url(./Sprite.png) no-repeat;
    background-size: 2500% 150px;
    animation: slow 0.8s steps(24) infinite;
    width: 150px;
    height: 150px;
    position: absolute;

    left: 50%;
    transform: translateX(-50%);
  }

  @keyframes slow {
    0% {
      background-position: 0 0;
    }
    100% {
      background-position: 100% 0;
    }
  }
```
