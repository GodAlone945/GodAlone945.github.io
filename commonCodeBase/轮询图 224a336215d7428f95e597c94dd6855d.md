```jsx
document.querySelector('.next').onclick = function() {
     
                    if (num <= length && num > 0) {
                        -- num
                        swiperWrapper.style.left = `-${num * 100}%`
                    } else if ( num <= 0) {
                        num = length - 1
                        swiperWrapper.style.left = `-${num * 100}%`
                    }
                }
                document.querySelector('.prev').onclick = function() {
                
                    if (num < length-1 && num >= 0) {
                        ++ num
                        swiperWrapper.style.left = `-${num * 100}%`
                    } else if ( num >= length - 1) {
                        num = 0
                        swiperWrapper.style.left = '0'
                    }
                }
```
