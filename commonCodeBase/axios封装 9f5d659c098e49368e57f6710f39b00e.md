```jsx
/**
 * 利用axios配置网络请求
 * @param config 请求需要的配置信息
 * @returns {*} 返回一个promise对象
 */
function request (config) {
  // 创建axios的实例
  const instance = axios.create({
    baseURL: 'http://zhfb.fenbanruanjian.com/division/',
    headers: {
      'Content-Type': 'application/json;charset=UTF-8'
    }
  })

  // 发送网络请求
  return instance(config)
}

// 调用方法
instance({
	method: post,
	url: 'divisionrecord/updateStatus',
	data: {
		ids: [1,2,3],
		status: 1
	}
})
.then((res) => {
	console.log(res)
})
.catch((err) => {
	console.log(err)
})
```
