# Echarts大小自适应

Created: July 9, 2021 11:28 AM
标签: Echarts, JS

```jsx
option = {
    xAxis: {
        type: 'category',
        data: ['Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat', 'Sun']
    },
    yAxis: {
        type: 'value'
    },
    series: [{
        data: [820, 932, 901, 934, 1290, 1330, 1320],
        type: 'line'
    }]
};
// 自适应的部分
myChart.clear();
        myChart.setOption(option,true);
        window.addEventListener('resize',() => {
            if(myChart){
                myChart.resize()
            }
        })
```
